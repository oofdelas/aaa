local Players = game:GetService("Players")
local UserInputService = game:GetService("UserInputService")

local LocalPlayer = Players.LocalPlayer
local PlayerGui = LocalPlayer:WaitForChild("PlayerGui")

local existing = PlayerGui:FindFirstChild("AlbuxLockScreen")
if existing then
    existing:Destroy()
end

local gui = Instance.new("ScreenGui")
gui.Name = "AlbuxLockScreen"
gui.IgnoreGuiInset = true
gui.ResetOnSpawn = false
gui.DisplayOrder = 999999
gui.Parent = PlayerGui

local bg = Instance.new("Frame")
bg.Size = UDim2.new(1, 0, 1, 0)
bg.Position = UDim2.new(0, 0, 0, 0)
bg.BackgroundColor3 = Color3.fromRGB(0, 0, 0)
bg.BorderSizePixel = 0
bg.Parent = gui

local text = Instance.new("TextLabel")
text.Size = UDim2.new(1, -40, 0, 160)
text.Position = UDim2.new(0, 20, 0.5, -80)
text.BackgroundTransparency = 1
text.TextColor3 = Color3.fromRGB(255, 255, 255)
text.Font = Enum.Font.GothamBlack
text.TextSize = 26
text.TextWrapped = true
text.TextYAlignment = Enum.TextYAlignment.Center
text.TextXAlignment = Enum.TextXAlignment.Center
text.Text = "IF U WANT USE ALBUX HUB, BUY IT!\n\ndiscord.gg/albuxhub"
text.Parent = bg

local stroke = Instance.new("UIStroke")
stroke.Color = Color3.fromRGB(0, 0, 0)
stroke.Transparency = 0.2
stroke.Thickness = 2
stroke.Parent = text

UserInputService.InputBegan:Connect(function(inp)
    if inp.UserInputType == Enum.UserInputType.Keyboard and inp.KeyCode == Enum.KeyCode.F8 then
        if gui and gui.Parent then
            gui:Destroy()
        end
    end
end)
